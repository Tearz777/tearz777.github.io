import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# ==============================
# LOAD DATA
# ==============================
df = pd.read_csv("Mlbb_Heroes.csv")

# CLEAN COLUMN
# ==============================
df.columns = df.columns.str.title()

# ==============================
# ROLE COUNT
# ==============================
role_counts = df["Primary_Role"].value_counts()

# ==============================
# PRINT HASIL
# ==============================
print("Distribusi Role:\n")
print(role_counts)

# ==============================
# VISUALISASI (BAR CHART)
# ==============================
plt.figure(figsize=(10,6))

role_counts.plot(kind="bar")

plt.title("Distribusi Role Hero MLBB")
plt.xlabel("Role")
plt.ylabel("Jumlah Hero")

plt.xticks(rotation=45)

plt.show()

from matplotlib.ticker import PercentFormatter
sns.set_theme()
# ==============================
# FEATURE ENGINEERING
# ==============================

# total game
df["games"] = df["Esport_Wins"] + df["Esport_Loss"]

# winrate
df["winrate"] = df["Esport_Wins"] / df["games"]

# ==============================
# FILTER (min 50 games)
# ==============================
df_filtered = df[df["games"] >= 50]

# ==============================
# TOP 10 HERO
# ==============================
top10 = df_filtered.sort_values(by="winrate", ascending=False).head(10)
# ==============================
# PLOT
# ==============================
plt.figure(figsize=(10,6))

sns.barplot(
    data=top10,
    x="winrate",
    y="Name"
)

plt.title("Top 10 Hero - Esport Win Rate (Min 50 Games)")
plt.xlabel("Win Rate")
plt.ylabel("Hero")
plt.xticks(rotation=45, ha='right')
plt.gca().xaxis.set_major_formatter(PercentFormatter(1.0))

plt.show()
